def _capture(logger: logging.Logger, formatter: logging.Formatter) -> io.StringIO:
    buf = io.StringIO()
    h = logging.StreamHandler(buf)
    h.setFormatter(formatter)
    logger.addHandler(h)
    logger.setLevel(logging.DEBUG)
    logger.propagate = False
    return buf
