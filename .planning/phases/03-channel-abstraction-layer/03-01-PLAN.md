```python
import asyncio
from typing import Optional
from .base import BaseChannel


class ChannelRegistry:
    def __init__(self):
        self._channels: dict[str, BaseChannel] = {}
        self._tasks: dict[str, asyncio.Task] = {}

    def register(self, channel: BaseChannel) -> None:
        """Register a channel. Raises ValueError if channel_id already registered."""
        cid = channel.channel_id
        if cid in self._channels:
            raise ValueError(f"Channel '{cid}' already registered")
        self._channels[cid] = channel

    def get(self, channel_id: str) -> Optional[BaseChannel]:
        """Return the channel or None if not registered."""
        return self._channels.get(channel_id)

    async def start_all(self) -> None:
        """
        Start all registered channels as asyncio tasks within the current event loop.
        MUST be called from inside an async context (FastAPI lifespan).
        Uses asyncio.create_task() — NEVER asyncio.run() — uvicorn already owns the event loop.
        """
        for cid, channel in self._channels.items():
            task = asyncio.create_task(channel.start(), name=f"channel-{cid}")
            self._tasks[cid] = task
            print(f"[CHANNELS] Started channel: {cid}")

    async def stop_all(self) -> None:
        """Cancel all channel tasks and call stop() on each channel."""
        for task in self._tasks.values():
            task.cancel()
        if self._tasks:
            await asyncio.gather(*self._tasks.values(), return_exceptions=True)
        self._tasks.clear()
        for cid, channel in self._channels.items():
            await channel.stop()
            print(f"[CHANNELS] Stopped channel: {cid}")

    def list_ids(self) -> list[str]:
        return list(self._channels.keys())
```

```python
from datetime import datetime
from .base import BaseChannel, ChannelMessage


class StubChannel(BaseChannel):
    """
    Minimal channel for testing and pattern validation.
    Does not connect to any external service.
    asyncio.create_task() wraps start() — returning immediately is correct for stubs.
    Real channels (Telegram, Discord) put their polling loop inside start().
    """

    def __init__(self, channel_id: str = "stub"):
        self._channel_id = channel_id
        self.sent_messages: list[tuple[str, str]] = []
        self._started = False

    @property
    def channel_id(self) -> str:
        return self._channel_id

    async def receive(self, raw_payload: dict) -> ChannelMessage:
        return ChannelMessage(
            channel_id=self._channel_id,
            user_id=raw_payload.get("user_id", "stub_user"),
            chat_id=raw_payload.get("chat_id", "stub_chat"),
            text=raw_payload.get("text", ""),
            message_id=raw_payload.get("message_id", ""),
            sender_name=raw_payload.get("sender_name", "Stub User"),
            raw=raw_payload,
        )

    async def send(self, chat_id: str, text: str) -> bool:
        self.sent_messages.append((chat_id, text))
        return True

    async def send_typing(self, chat_id: str) -> None:
        pass

    async def mark_read(self, chat_id: str, message_id: str) -> None:
        pass

    async def health_check(self) -> dict:
        return {"status": "ok", "channel": self._channel_id, "started": self._started}

    async def start(self) -> None:
        self._started = True
        # Returns immediately — asyncio.create_task() schedules this as a task.
        # Real polling channels would loop here: while True: await asyncio.sleep(...)

    async def stop(self) -> None:
        self._started = False
```
