    return {
        "status": "configured" if has_providers else "unconfigured",
        "provider": "cloud",
        "model": casual_model,
    }

    monkeypatch.setenv("SYNAPSE_HOME", str(tmp_path))
    # Force fresh module state
    for mod in list(sys.modules.keys()):

    conn = sqlite3.connect(db_mod.DB_PATH)
    tables = [r[0] for r in conn.execute(
        "SELECT name FROM sqlite_master WHERE type='table'"

def test_no_openclaw_binary_calls_in_active_files():
    """Phase 7 success criterion: no openclaw binary calls in non-migration workspace files."""
    workspace = os.path.join(os.path.dirname(__file__), "..")
    # Files that are ALLOWED to contain 'openclaw' (migration helpers, ONB-08)
    allowed_patterns = [
        "scripts/migrate_openclaw",
        "scripts/v2_migration",
        "cli/onboard",
        "tests/test_migration",
        "tests/test_onboard",
    ]
    violations = []
    for root, dirs, files in os.walk(workspace):
        # Skip __pycache__ and .git
        dirs[:] = [d for d in dirs if d not in ("__pycache__", ".git")]
        for fname in files:
            if not fname.endswith(".py"):
                continue
            fpath = os.path.join(root, fname)
            relpath = os.path.relpath(fpath, workspace).replace("\\", "/")
            if any(pat in relpath for pat in allowed_patterns):
                continue
            content = open(fpath, encoding="utf-8", errors="ignore").read()
            if "openclaw" in content:
                violations.append(relpath)
    assert not violations, (
        f"Unexpected openclaw references in: {violations}\n"
        "Only migrate_openclaw.py, v2_migration/, cli/onboard.py, test_migration.py, "
        "test_onboard.py may contain 'openclaw'."
