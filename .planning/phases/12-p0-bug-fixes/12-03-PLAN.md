        cfg = SynapseConfig.load()
        session_cfg = getattr(cfg, "session", {}) or {}

        for user_id, channel_id in [("the_creator", "whatsapp"), ("the_partner", "whatsapp")]:
            try:
                reply = await self.proactive_engine.maybe_reach_out(user_id, channel_id)
