Top-level entry point called from synapse_cli.py. Structure:
```python
def run_wizard(non_interactive: bool = False) -> None:
    """Entry point — dispatches to interactive or non-interactive wizard."""
    if non_interactive or not _is_tty():
        _run_non_interactive()
    else:
        _run_interactive()

def _is_tty() -> bool:
    try:
        return sys.stdin.isatty()
    except AttributeError:
        return False
```

**3. _run_non_interactive() -> None**

Logic:
1. `provider = os.environ.get("SYNAPSE_PRIMARY_PROVIDER", "").strip()` — if empty: `typer.echo("ERROR: SYNAPSE_PRIMARY_PROVIDER is required ...", err=True); raise typer.Exit(1)`
2. If `provider not in PROVIDER_LIST`: `typer.echo(f"ERROR: Unknown provider '{provider}'. Valid: {', '.join(PROVIDER_LIST)}", err=True); raise typer.Exit(1)`
3. Determine data_root: `data_root = Path(os.environ.get("SYNAPSE_HOME", Path.home() / ".synapse"))`
4. Build config dict:
   ```python
   config: dict = {"providers": {}, "model_mappings": {}, "channels": {}}
   ```
5. For providers: if `provider == "ollama"`: read `OLLAMA_API_BASE` env var (default "http://localhost:11434"); validate with validate_ollama(); add to config. Else: get `env_var = _KEY_MAP[provider]`; `api_key = os.environ.get(env_var, "").strip()` — if empty: `typer.echo(f"ERROR: {env_var} is required for provider '{provider}' ...", err=True); raise typer.Exit(1)`. Run validate_provider(). If not ok and error != "quota_exceeded": `typer.echo(f"ERROR: Provider validation failed: {result.error} — {result.detail}", err=True); raise typer.Exit(1)`.
6. Channel secrets (all optional in NI mode — skip if env var absent):
   - `tg_token = os.environ.get("SYNAPSE_TELEGRAM_TOKEN")` → if set: `config["channels"]["telegram"] = {"token": tg_token}`
   - `ds_token = os.environ.get("SYNAPSE_DISCORD_TOKEN")` → if set: `config["channels"]["discord"] = {"token": ds_token, "allowed_channel_ids": []}`
   - `slk_bot = os.environ.get("SYNAPSE_SLACK_BOT_TOKEN"); slk_app = os.environ.get("SYNAPSE_SLACK_APP_TOKEN")` → if both set: `config["channels"]["slack"] = {"bot_token": slk_bot, "app_token": slk_app}`
7. Generate model_mappings defaults via `_build_model_mappings(configured_providers: list[str])`.
8. `write_config(data_root, config)`; `typer.echo(f"Config written to {data_root / 'synapse.json'}")`

```python
def _build_model_mappings(providers: list[str]) -> dict:
    """Generate sensible model_mappings based on which providers were configured."""
    mappings = {}
    # casual: gemini > openai > groq > first available
    for cand, model in [
        ("gemini",     "gemini/gemini-2.0-flash"),
        ("openai",     "openai/gpt-4o-mini"),
        ("groq",       "groq/llama-3.3-70b-versatile"),
        ("anthropic",  "anthropic/claude-haiku-4-5"),
    ]:
        if cand in providers:
            mappings["casual"] = {"model": model, "fallback": None}
            break
    # code: anthropic > openai > groq
    for cand, model in [
        ("anthropic",  "anthropic/claude-sonnet-4-6"),
        ("openai",     "openai/gpt-4o"),
        ("groq",       "groq/llama-3.3-70b-versatile"),
    ]:
        if cand in providers:
            mappings["code"] = {"model": model, "fallback": None}
            break
    # vault: always ollama (local-only by design)
    if "ollama" in providers:
        mappings["vault"] = {"model": "ollama_chat/llama3.3", "fallback": None}
    return mappings
```

def _run_migration(openclaw_root: Path, dest_root: Path) -> None:
    try:
        import importlib.util, sys
        spec = importlib.util.spec_from_file_location(
            "migrate_openclaw",
            Path(__file__).resolve().parent.parent / "scripts" / "migrate_openclaw.py",
        )
        mod = importlib.util.module_from_spec(spec)
        spec.loader.exec_module(mod)
        mod.migrate(src=openclaw_root, dest=dest_root)
        console.print("[green]Migration complete.[/]")
    except Exception as exc:
        console.print(f"[red]Migration failed: {exc}[/]")
